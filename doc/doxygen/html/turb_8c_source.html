135 static void pspect(ath_fft_data *ampl)
136 {
137  int i,j,k;
138  double q1,q2,q3;
139 
140  /* set random amplitudes with gaussian deviation */
141  for (k=0; k<nx3; k++) {
142  for (j=0; j<nx2; j++) {
143  for (i=0; i<nx1; i++) {
144  q1 = ran2(&rseed);
145  q2 = ran2(&rseed);
146  q3 = sqrt(-2.0*log(q1+1.0e-20))*cos(2.0*PI*q2);
147  q1 = ran2(&rseed);
148  ampl[OFST(i,j,k)][0] = q3*cos(2.0*PI*q1);
149  ampl[OFST(i,j,k)][1] = q3*sin(2.0*PI*q1);
150  }
151  }
152  }
153 
154  /* set power spectrum
155  * ispect=1: power law - original form
156  * ispect=2: form from Gammie&Ostriker
157  */
158  for (k=0; k<nx3; k++) {
159  for (j=0; j<nx2; j++) {
160  for (i=0; i<nx1; i++) {
161  /* compute k/dkx */
162  q3 = KWVM(i,j,k);
163  if ((q3 > klow) && (q3 < khigh)) {
164  q3 *= dkx; /* multiply by 2 pi/L */
165  if (ispect == 1) {
166  /* decreasing power law */
167  ampl[OFST(i,j,k)][0] /= pow(q3,(expo+2.0)/2.0);
168  ampl[OFST(i,j,k)][1] /= pow(q3,(expo+2.0)/2.0);
169  } else if (ispect == 2) {
170  /* G&O form */
171  ampl[OFST(i,j,k)][0] *= pow(q3,3.0)*exp(-4.0*q3/kpeak);
172  ampl[OFST(i,j,k)][1] *= pow(q3,3.0)*exp(-4.0*q3/kpeak);
173  }
174  } else {
175  /* introduce cut-offs at klow and khigh */
176  ampl[OFST(i,j,k)][0] = 0.0;
177  ampl[OFST(i,j,k)][1] = 0.0;
178  }
179  }
180  }
181  }
182  ampl[0][0] = 0.0;
183  ampl[0][1] = 0.0;
184 
185  return;
186 }

193 static void project()
194 {
195  int i,j,k,m,ind;
196  double kap[3], kapn[3], mag;
197  ath_fft_data dot;
198 
199  /* Project off non-solenoidal component of velocity */
200  for (k=0; k<nx3; k++) {
201  kap[2] = sin(2.0*PI*(gks+k)/gnx3);
202  for (j=0; j<nx2; j++) {
203  kap[1] = sin(2.0*PI*(gjs+j)/gnx2);
204  for (i=0; i<nx1; i++) {
205  if (((gis+i)+(gjs+j)+(gks+k)) != 0) {
206  kap[0] = sin(2.0*PI*(gis+i)/gnx1);
207  ind = OFST(i,j,k);
208 
209  /* make kapn a unit vector */
210  mag = sqrt(SQR(kap[0]) + SQR(kap[1]) + SQR(kap[2]));
211  for (m=0; m<3; m++) kapn[m] = kap[m] / mag;
212 
213  /* find fv_0 dot kapn */
214  dot[0] = fv1[ind][0]*kapn[0]+fv2[ind][0]*kapn[1]+fv3[ind][0]*kapn[2];
215  dot[1] = fv1[ind][1]*kapn[0]+fv2[ind][1]*kapn[1]+fv3[ind][1]*kapn[2];
216 
217  /* fv = fv_0 - (fv_0 dot kapn) * kapn */
218  fv1[ind][0] -= dot[0]*kapn[0];
219  fv2[ind][0] -= dot[0]*kapn[1];
220  fv3[ind][0] -= dot[0]*kapn[2];
221 
222  fv1[ind][1] -= dot[1]*kapn[0];
223  fv2[ind][1] -= dot[1]*kapn[1];
224  fv3[ind][1] -= dot[1]*kapn[2];
225  }
226  }
227  }
228  }
229 
230  return;
231 }

280 static void perturb(Grid *pGrid, Real dt)
281 {
282  int i, is=pGrid->is, ie = pGrid->ie;
283  int j, js=pGrid->js, je = pGrid->je;
284  int k, ks=pGrid->ks, ke = pGrid->ke;
285  int ind, mpierr;
286  Real dvol, aa, b, c, s, de, qa, v1, v2, v3;
287  Real t0, t0ij, t0i, t1, t1ij, t1i;
288  Real t2, t2ij, t2i, t3, t3ij, t3i;
289  Real m[4], gm[4];
290 
291  /* Set the velocities in real space */
292  dvol = 1.0/((Real)(gnx1*gnx2*gnx3));
293  for (k=ks; k<=ke; k++) {
294  for (j=js; j<=je; j++) {
295  for (i=is; i<=ie; i++) {
296  ind = OFST(i-is,j-js,k-ks);
297  dv1[k][j][i] = fv1[ind][0]*dvol;
298  dv2[k][j][i] = fv2[ind][0]*dvol;
299  dv3[k][j][i] = fv3[ind][0]*dvol;
300  }
301  }
302  }
303 
304  /* Calculate net momentum pertubation components t1, t2, t3 */
305  t0 = 0.0; t1 = 0.0; t2 = 0.0; t3 = 0.0;
306  for (k=ks; k<=ke; k++) {
307  t0ij = 0.0; t1ij = 0.0; t2ij = 0.0; t3ij = 0.0;
308  for (j=js; j<=je; j++) {
309  t0i = 0.0; t1i = 0.0; t2i = 0.0; t3i = 0.0;
310  for (i=is; i<=ie; i++) {
311  t0i += pGrid->U[k][j][i].d;
312 
313  /* The net momentum perturbation */
314  t1i += pGrid->U[k][j][i].d * dv1[k][j][i];
315  t2i += pGrid->U[k][j][i].d * dv2[k][j][i];
316  t3i += pGrid->U[k][j][i].d * dv3[k][j][i];
317  }
318  t0ij += t0i; t1ij += t1i; t2ij += t2i; t3ij += t3i;
319  }
320  t0 += t0ij; t1 += t1ij; t2 += t2ij; t3 += t3ij;
321  }
322 
323 #ifdef MPI_PARALLEL
324  /* Sum the perturbations over all processors */
325  m[0] = t0; m[1] = t1; m[2] = t2; m[3] = t3;
326  mpierr = MPI_Allreduce(m, gm, 4, MPI_RL, MPI_SUM, MPI_COMM_WORLD);
327  if (mpierr) ath_error("[normalize]: MPI_Allreduce error = %d\n", mpierr);
328  t0 = gm[0]; t1 = gm[1]; t2 = gm[2]; t3 = gm[3];
329 #endif /* MPI_PARALLEL */
330 
331  /* Subtract the mean velocity perturbation so that the net momentum
332  * perturbation is zero. */
333  for (k=ks; k<=ke; k++) {
334  for (j=js; j<=je; j++) {
335  for (i=is; i<=ie; i++) {
336  dv1[k][j][i] -= t1/t0;
337  dv2[k][j][i] -= t2/t0;
338  dv3[k][j][i] -= t3/t0;
339  }
340  }
341  }
342 
343  /* Calculate unscaled energy of perturbations */
344  t1 = 0.0; t2 = 0.0;
345  for (k=ks; k<=ke; k++) {
346  t1ij = 0.0; t2ij = 0.0;
347  for (j=js; j<=je; j++) {
348  t1i = 0.0; t2i = 0.0;
349  for (i=is; i<=ie; i++) {
350  /* Calculate velocity pertubation at cell center from
351  * perturbations at cell faces */
352  v1 = dv1[k][j][i];
353  v2 = dv2[k][j][i];
354  v3 = dv3[k][j][i];
355 
356  t1i += (pGrid->U[k][j][i].d)*(SQR(v1) + SQR(v2) + SQR(v3));
357  t2i += (pGrid->U[k][j][i].M1)*v1 + (pGrid->U[k][j][i].M2)*v2 +
358  (pGrid->U[k][j][i].M3)*v3;
359  }
360  t1ij += t1i; t2ij += t2i;
361  }
362  t1 += t1ij; t2 += t2ij;
363  }
364 
365 #ifdef MPI_PARALLEL
366  /* Sum the perturbations over all processors */
367  m[0] = t1; m[1] = t2;
368  mpierr = MPI_Allreduce(m, gm, 2, MPI_RL, MPI_SUM, MPI_COMM_WORLD);
369  if (mpierr) ath_error("[normalize]: MPI_Allreduce error = %d\n", mpierr);
370  t1 = gm[0]; t2 = gm[1];
371 #endif /* MPI_PARALLEL */
372 
373  /* Rescale to give the correct energy injection rate */
374  dvol = pGrid->dx1*pGrid->dx2*pGrid->dx3;
375  if (idrive == 0) {
376  /* driven turbulence */
377  de = dedt*dt;
378  } else {
379  /* decaying turbulence (all in one shot) */
380  de = dedt;
381  }
382  aa = 0.5*t1;
383  aa = MAX(aa,1.0e-20);
384  b = t2;
385  c = -de/dvol;
386  if(b >= 0.0)
387  s = (-2.0*c)/(b + sqrt(b*b - 4.0*aa*c));
388  else
389  s = (-b + sqrt(b*b - 4.0*aa*c))/(2.0*aa);
390 
391  if (isnan(s)) ath_error("[perturb]: s is NaN!\n");
392 
393  /* Apply momentum pertubations */
394  for (k=ks; k<=ke; k++) {
395  for (j=js; j<=je; j++) {
396  for (i=is; i<=ie; i++) {
397  qa = s*pGrid->U[k][j][i].d;
398  pGrid->U[k][j][i].M1 += qa*dv1[k][j][i];
399  pGrid->U[k][j][i].M2 += qa*dv2[k][j][i];
400  pGrid->U[k][j][i].M3 += qa*dv3[k][j][i];
401  }
402  }
403  }
404 
405  return;
406 }
