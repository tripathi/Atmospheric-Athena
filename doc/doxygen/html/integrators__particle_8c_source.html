105  {
106  case 1: /* 2nd order explicit integrator */
107  int_par_exp(pG,curG,cell1, &dv1,&dv2,&dv3, &ts);
108  break;
109 
110  case 2: /* 2nd order semi-implicit integrator */
111  int_par_semimp(pG,curG,cell1, &dv1,&dv2,&dv3, &ts);

174  curG->x2 = curP->x2;
175  curG->x3 = curP->x3;

180 
181  } /* end of the for loop */
182

184  ath_pout(0, "In processor %d, there are %ld particles.\n",
185  pG->my_id, pG->nparticle);
186

201  Real *dv1, Real *dv2, Real *dv3, Real *ts)

210 #endif

248  /* shortcut expressions */
249  b0 = 1.0+pG->dt*ts11;
250

252  ft.x1 = 0.5*(fc.x1+b0*fp.x1);
253  ft.x2 = 0.5*(fc.x2+b0*fp.x2);
254  ft.x3 = 0.5*(fc.x3+b0*fp.x3);
255 
256 #ifdef SHEARING_BOX
257  oh = Omega_0*pG->dt;
258  if (pG->Nx3 > 1) {/* 3D shearing sheet (x1,x2,x3)=(X,Y,Z) */
259  ft.x1 += -oh*fp.x2;
260  #ifdef FARGO
261  ft.x2 += 0.5*(2.0-qshear)*oh*fp.x1;
262  #else
263  ft.x2 += oh*fp.x1;
264  #endif
265  } else { /* 2D shearing sheet (x1,x2,x3)=(X,Z,Y) */
266  ft.x1 += -oh*fp.x3;
267  #ifdef FARGO
268  ft.x3 += 0.5*(2.0-qshear)*oh*fp.x1;
269  #else
270  ft.x3 += oh*fp.x1;
271  #endif
272  }
273 #endif /* SHEARING_BOX */
274 
275  /* calculate the inverse matrix elements */
276  D = 1.0+0.5*pG->dt*(ts11 + ts12 + pG->dt*ts11*ts12);
277 #ifdef SHEARING_BOX
278  oh2 = SQR(oh);
279  B = oh * (-2.0-(ts11+ts12)*pG->dt);
280 #ifdef FARGO
281  A = D - (2.0-qshear)*oh2;
282  C = 0.5*(qshear-2.0)*B;
283 #else /* FARGO */
284  A = D - 2.0*oh2;
285  C = -B;
286 #endif /* FARGO */
287  Det1 = 1.0/(SQR(A)-B*C);
288  if (pG->Nx3>1) {
289  *dv1 = pG->dt*Det1*(ft.x1*A-ft.x2*B);
290  *dv2 = pG->dt*Det1*(-ft.x1*C+ft.x2*A);
291  *dv3 = pG->dt*ft.x3/D;
292  } else {
293  *dv1 = pG->dt*Det1*(ft.x1*A-ft.x3*B);
294  *dv3 = pG->dt*Det1*(-ft.x1*C+ft.x3*A);
295  *dv2 = pG->dt*ft.x2/D;
296  }
297 #else /* SHEARING_BOX */
298  D = 1.0/D;
299  *dv1 = pG->dt*ft.x1*D;
300  *dv2 = pG->dt*ft.x2*D;
301  *dv3 = pG->dt*ft.x3*D;
302 #endif /* SHEARING_BOX */
303 
304  *ts = 0.5/ts11+0.5/ts12;
305 
306  return;
307 }

749 Vector Get_Drag(Grid *pG, int type, Real x1, Real x2, Real x3,
750  Real v1, Real v2, Real v3, Vector cell1, Real *tstop1)
751 {
752  int is, js, ks;
753  Real rho, u1, u2, u3, cs;
754  Real vd1, vd2, vd3, vd, tstop, ts1;
755 #ifdef FEEDBACK
756  Real stiffness;
757 #endif
758  Real weight[3][3][3]; /* weight function */
759  Vector fd;
760 
761  /* interpolation to get fluid density, velocity and the sound speed */
762  getweight(pG, x1, x2, x3, cell1, weight, &is, &js, &ks);
763 
764 #ifndef FEEDBACK
765  if (getvalues(pG, weight, is, js, ks, &rho, &u1, &u2, &u3, &cs) == 0)
766 #else
767  if (getvalues(pG, weight, is, js, ks, &rho,&u1,&u2,&u3,&cs, &stiffness) == 0)
768 #endif
769  { /* particle in the grid */
770 
771  /* apply possible gas velocity shift (e.g., for fake gas velocity field) */
772  gasvshift(x1, x2, x3, &u1, &u2, &u3);
773 
774  /* velocity difference */
775  vd1 = v1-u1;
776  vd2 = v2-u2;
777  vd3 = v3-u3;
778  vd = sqrt(SQR(vd1) + SQR(vd2) + SQR(vd3)); /* dimension independent */
779 
780  /* particle stopping time */
781  tstop = get_ts(pG, type, rho, cs, vd);
782 #ifdef FEEDBACK
783 // tstop *= MAX(1.0,stiffness);
784 #endif
785  ts1 = 1.0/tstop;
786  }
787  else
788  { /* particle out of the grid, free motion, with warning sign */
789  vd1 = 0.0; vd2 = 0.0; vd3 = 0.0; ts1 = 0.0;
790  ath_perr(0, "Particle move out of grid %d with position (%f,%f,%f)!\n",
791  pG->my_id,x1,x2,x3); /* warning! */
792  }
793 
794  *tstop1 = ts1;
795 
796  /* Drag force */
797  fd.x1 = -ts1*vd1;
798  fd.x2 = -ts1*vd2;
799  fd.x3 = -ts1*vd3;
800 
801  return fd;
