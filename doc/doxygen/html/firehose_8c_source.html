245 double ran2(long int *idum)
246 {
247  int j;
248  long int k;
249  static long int idum2=123456789;
250  static long int iy=0;
251  static long int iv[NTAB];
252  double temp;
253 
254  if (*idum <= 0) { /* Initialize */
255  if (-(*idum) < 1) *idum=1; /* Be sure to prevent idum = 0 */
256  else *idum = -(*idum);
257  idum2=(*idum);
258  for (j=NTAB+7;j>=0;j--) { /* Load the shuffle table (after 8 warm-ups) */
259  k=(*idum)/IQ1;
260  *idum=IA1*(*idum-k*IQ1)-k*IR1;
261  if (*idum < 0) *idum += IM1;
262  if (j < NTAB) iv[j] = *idum;
263  }
264  iy=iv[0];
265  }
266  k=(*idum)/IQ1; /* Start here when not initializing */
267  *idum=IA1*(*idum-k*IQ1)-k*IR1; /* Compute idum=(IA1*idum) % IM1 without */
268  if (*idum < 0) *idum += IM1; /* overflows by Schrage's method */
269  k=idum2/IQ2;
270  idum2=IA2*(idum2-k*IQ2)-k*IR2; /* Compute idum2=(IA2*idum) % IM2 likewise */
271  if (idum2 < 0) idum2 += IM2;
272  j=(int)(iy/NDIV); /* Will be in the range 0...NTAB-1 */
273  iy=iv[j]-idum2; /* Here idum is shuffled, idum and idum2 */
274  iv[j] = *idum; /* are combined to generate output */
275  if (iy < 1) iy += IMM1;
276  if ((temp=AM*iy) > RNMX) return RNMX; /* No endpoint values */
277  else return temp;
278 }
