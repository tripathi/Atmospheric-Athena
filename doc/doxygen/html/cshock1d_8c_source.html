260 Real RK4(Real D, Real A, Real M, Real theta, Real h)
261 {
262  Real k1, k2, k3, k4;
263 
264  k1 = Dprime(D,A,M,theta);
265  k2 = Dprime(D+0.5*h*k1,A,M,theta);
266  k3 = Dprime(D+0.5*h*k2,A,M,theta);
267  k4 = Dprime(D+h*k3,A,M,theta);
268 
269  return D + h/6.0*(k1+2.0*k2+2.0*k3+k4);
270 
271 }

275 Real Dprime(Real D, Real A, Real M, Real theta)
276 {
277  Real sintheta, costheta, sintheta2, costheta2;
278  Real M21, b, b2;
279 
280  sintheta = sin(theta);
281  costheta = cos(theta);
282  sintheta2 = SQR(sintheta);
283  costheta2 = SQR(costheta);
284  M21 = 1.0/SQR(M);
285 
286  b2 = sintheta2 + 2*SQR(A)*(D-1.0)*(1.0/D-M21);
287  b = sqrt(b2);
288 
289  return b/A*(b-D*((b-sintheta)/SQR(A)*costheta2+sintheta))/(b2+costheta2)/(1/SQR(D)-M21);
290 }
