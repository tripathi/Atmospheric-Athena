724 void ShearingSheet_ox1(DomainS *pD)
725 {
726  GridS *pG = pD->Grid;
727  int is = pG->is, ie = pG->ie;
728  int js = pG->js, je = pG->je;
729  int ks = pG->ks, ke = pG->ke;
730  int i,ii,j,k,ku,n,joffset,jremap;
731  Real xmin,xmax,Lx,Ly,qomL,yshear,deltay,epso;
732 #ifdef MPI_PARALLEL
733  int my_iproc,my_jproc,my_kproc,cnt,jproc,joverlap,Ngrids;
734  int ierr,sendto_id,getfrom_id;
735  double *pSnd,*pRcv;
736  Remap *pRemap;
737  ConsS *pCons;
738  MPI_Request rq;
739 #endif
740 
741 /*--- Step 1. ------------------------------------------------------------------
742  * Compute the distance the computational domain has sheared in y */
743 
744  xmin = pD->RootMinX[0];
745  xmax = pD->RootMaxX[0];
746  Lx = xmax - xmin;
747 
748  xmin = pD->RootMinX[1];
749  xmax = pD->RootMaxX[1];
750  Ly = xmax - xmin;
751 
752  qomL = qshear*Omega_0*Lx;
753  yshear = qomL*pG->time;
754 
755 /* Split this into integer and fractional peices of the Domain in y. Ignore
756  * the integer piece because the Grid is periodic in y */
757 
758  deltay = fmod(yshear, Ly);
759 
760 /* further decompose the fractional peice into integer and fractional pieces of
761  * a grid cell. Note 0.0 <= epsi < 1.0. If Domain has MPI decomposition in Y,
762  * then it is possible that: pD->Nx2 > joffset > pG->Nx2 */
763 
764  joffset = (int)(deltay/pG->dx2);
765  epso = -(fmod(deltay,pG->dx2))/pG->dx2;
766 
767 /*--- Step 2. ------------------------------------------------------------------
768  * Copy data into GhstZns array. Note i and j indices are switched.
769  * Steps 2-10 are for 3D or 2D xy runs. Step 11 handles 2D xz separately */
770 
771  if (pG->Nx[2] > 1 || ShBoxCoord==xy) { /* this if ends at end of step 10 */
772 
773  if (pG->Nx[2] > 1) ku=ke+1; else ku=ke;
774  for(k=ks; k<=ku; k++) {
775  for(j=js-nghost; j<=je+nghost; j++){
776  for(i=0; i<nghost; i++){
777  ii = ie+1+i;
778  GhstZns[k][i][j].U[0] = pG->U[k][j][ii].d;
779  GhstZns[k][i][j].U[1] = pG->U[k][j][ii].M1;
780  GhstZns[k][i][j].U[2] = pG->U[k][j][ii].M2;
781 #ifndef FARGO
782  GhstZns[k][i][j].U[2] -= qomL*pG->U[k][j][ii].d;
783 #endif
784  GhstZns[k][i][j].U[3] = pG->U[k][j][ii].M3;
785 #ifdef ADIABATIC
786 /* No change in the internal energy */
787  GhstZns[k][i][j].U[4] = pG->U[k][j][ii].E + (0.5/GhstZns[k][i][j].U[0])*
788  (SQR(GhstZns[k][i][j].U[2]) - SQR(pG->U[k][j][ii].M2));
789 #endif /* ADIABATIC */
790 #ifdef MHD
791  GhstZns[k][i][j].U[NREMAP-4] = pG->U[k][j][ii].B1c;
792  GhstZns[k][i][j].U[NREMAP-3] = pG->B1i[k][j][ii];
793  GhstZns[k][i][j].U[NREMAP-2] = pG->B2i[k][j][ii];
794  GhstZns[k][i][j].U[NREMAP-1] = pG->B3i[k][j][ii];
795 #endif /* MHD */
796 #if (NSCALARS > 0)
797  for(n=0; n<NSCALARS; n++) GhstZns[k][i][j].s[n] = pG->U[k][j][ii].s[n];
798 #endif
799  }
800  }
801  }
802 
803 /*--- Step 3. ------------------------------------------------------------------
804  * Copy GhstZns into buffer, at the same time apply a conservative remap of
805  * solution over the fractional part of grid cell */
806 
807  for(k=ks; k<=ku; k++) {
808  for(i=0; i<nghost; i++){
809 
810  for (n=0; n<(NREMAP); n++) {
811  for (j=js-nghost; j<=je+nghost; j++) U[j] = GhstZns[k][i][j].U[n];
812  RemapFlux(U,epso,js,je+1,Flx);
813  for(j=js; j<=je; j++){
814  GhstZnsBuf[k][i][j].U[n] = GhstZns[k][i][j].U[n] - (Flx[j+1]-Flx[j]);
815  }
816  }
817 
818 #if (NSCALARS > 0)
819  for (n=0; n<(NSCALARS); n++) {
820  for (j=js-nghost; j<=je+nghost; j++) U[j] = GhstZns[k][i][j].s[n];
821  RemapFlux(U,epso,js,je+1,Flx);
822  for(j=js; j<=je; j++){
823  GhstZnsBuf[k][i][j].s[n] = GhstZns[k][i][j].s[n] - (Flx[j+1]-Flx[j]);
824  }
825  }
826 #endif
827 
828  }
829  }
830 
831 /*--- Step 4. ------------------------------------------------------------------
832  * If no MPI decomposition in Y, apply shift over integer number of
833  * grid cells during copy from buffer back into GhstZns. */
834 
835  if (pD->NGrid[1] == 1) {
836 
837  for(k=ks; k<=ku; k++) {
838  for(j=js; j<=je; j++){
839  jremap = j + joffset;
840  if (jremap > (int)je) jremap -= pG->Nx[1];
841 
842  for(i=0; i<nghost; i++){
843  for (n=0; n<(NREMAP); n++) {
844  GhstZns[k][i][j].U[n] = GhstZnsBuf[k][i][jremap].U[n];
845  }
846 #if (NSCALARS > 0)
847  for (n=0; n<NSCALARS; n++) {
848  GhstZns[k][i][j].s[n] = GhstZnsBuf[k][i][jremap].s[n];
849  }
850 #endif
851  }
852 
853  }
854  }
855 
856 #ifdef MPI_PARALLEL
857  } else {
858 
859 /*--- Step 5. ------------------------------------------------------------------
860  * If Domain contains MPI decomposition in Y, then MPI calls are required for
861  * the cyclic shift needed to apply shift over integer number of grid cells
862  * during copy from buffer back into GhstZns. */
863 
864  get_myGridIndex(pD, myID_Comm_world, &my_iproc, &my_jproc, &my_kproc);

867  * This assumes every grid has same number of cells in x2-direction! */

880  if (jproc < 0) jproc += pD->NGrid[1];

884  if (jproc > (pD->NGrid[1]-1)) jproc -= pD->NGrid[1];
885  getfrom_id = pD->GData[my_kproc][jproc][my_iproc].ID_Comm_Domain;

887 /*--- Step 5b. -----------------------------------------------------------------
888  * Pack send buffer and send data in [js:js+(joverlap-1)] from GhstZnsBuf */

890  cnt = nghost*joverlap*(ku-ks+1)*(NREMAP+NSCALARS);
891 /* Post a non-blocking receive for the input data */
892  ierr = MPI_Irecv(recv_buf, cnt, MPI_DOUBLE, getfrom_id,
893  shearing_sheet_ox1_tag, pD->Comm_Domain, &rq);

896  for (k=ks; k<=ku; k++) {
897  for (j=js; j<=js+(joverlap-1); j++) {
898  for(i=0; i<nghost; i++){
899  /* Get a pointer to the Remap structure */
900  pRemap = &(GhstZnsBuf[k][i][j]);
901 
902  for (n=0; n<NREMAP; n++) *(pSnd++) = pRemap->U[n];
903 #if (NSCALARS > 0)
904  for (n=0; n<NSCALARS; n++) *(pSnd++) = pRemap->s[n];
905 #endif
906  }
907  }

919

923  for(i=0; i<nghost; i++){

932  }
933  }

938  * If shear is less one full Grid, remap cells which remain on same processor
939  * from GhstZnsBuf into GhstZns. Cells in [js+joverlap:je] are shifted by
940  * joverlap into [js:je-joverlap] */

943 
944  for(k=ks; k<=ku; k++) {
945  for(j=js; j<=je-joverlap; j++){
946  jremap = j+joverlap;
947  for(i=0; i<nghost; i++){
948  for (n=0; n<(NREMAP); n++) {

955 #endif

957  }

961  * If shear is more than one Grid, pack and send data from [js+joverlap:je]

963 
964  } else {

1005  for(i=0; i<nghost; i++){

1007  pRemap = &(GhstZns[k][i][j]);
1008  for (n=0; n<NREMAP; n++) pRemap->U[n] = *(pRcv++);
1009 #if (NSCALARS > 0)
1010  for (n=0; n<NSCALARS; n++) pRemap->s[n] = *(pRcv++);
1011 #endif
1012  }
1013  }
1014  }
1015  } /* end of step 5e - shear is more than one Grid */
1016 
1017 #endif /* MPI_PARALLEL */
1018  } /* end of step 5 - MPI decomposition in Y */
1019

1022

1038 #endif /* MHD */
