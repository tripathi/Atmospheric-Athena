77 
78  if(!(Ul.d > 0.0)||!(Ur.d > 0.0))
79  ath_error("[exact flux]: Non-positive densities: dl = %e dr = %e\n",
80  Ul.d, Ur.d);

82 /*--- Step 1. ------------------------------------------------------------------
83  * Compute the density and momentum of the intermediate state
84  */
85 
86  zl = sqrt((double)Wl.d);
87  zr = sqrt((double)Wr.d);
88 
89  /* --- 1-shock and 2-shock --- */
90  soln = 0;
91 
92  /* Start by finding density if shocks on both left and right.
93  * This will only be the case if dm > Wl.d and dm > Wr.d */
94  tmp = zl*zr*(Wl.Vx - Wr.Vx)/(2.0*Iso_csound*(zl + zr));
95  zm = tmp + sqrt((double)(tmp*tmp + zl*zr));
96  dm = zm*zm;
97 
98  /* Get velocity from 1-shock formula */
99  Vxm = Wl.Vx - Iso_csound*(dm-Wl.d)/(zm*zl);
100 
101  /* If left or right density is greater than intermediate density,
102  * then at least one side has rarefaction instead of shock */
103  dmin = MIN(Wl.d, Wr.d);
104  dmax = MAX(Wl.d, Wr.d);
105  if (dm < dmax) {
106  /* --- 1-rarefaction and 2-rarefaction --- */
107  soln = 3;
108 
109  /* Try rarefactions on both left and right, since it's a quicker
110  * calculation than 1-shock+2-raref or 1-raref+2-shock */
111  dm = zl*zr*exp((Wl.Vx-Wr.Vx)/(2.0*Iso_csound));
112 
113  /* Get velocity from 1-rarefaction formula */
114  Vxm = Wl.Vx - Iso_csound*log(dm/Wl.d);
115 
116  /* If left or right density is smaller than intermediate density,
117  * then we must instead have a combination of shock and rarefaction */
118  if (dm > dmin) {
119  /* --- EITHER 1-rarefaction and 2-shock ---
120  * --- OR 1-shock and 2-rarefaction --- */
121 
122  /* Solve iteratively equation for shock and rarefaction
123  * If Wl.d > Wr.d ==> 1-rarefaction and 2-shock
124  * If Wr.d > Wl.d ==> 1-shock and 2-rarefaction */
125  if (Wl.d > Wr.d) soln = 2; else soln = 1;
126 
127  dm = rtsafe(&srder,dmin,dmax, 2.0*DBL_EPSILON, Wl.Vx, Wr.Vx, dmin, dmax);
128 
129  /* Don't be foolish enough to take ln of zero */
130  if ((dm > dmin) && (dm <= dmax)) {
131  if (Wl.d > Wr.d) {
132  /* Get velocity from 1-rarefaction formula */
133  Vxm = Wl.Vx - Iso_csound*log(dm/Wl.d);
134  } else {
135  /* Get velocity from 2-rarefaction formula */
136  Vxm = Wr.Vx + Iso_csound*log(dm/Wr.d);
137  }
138  } else {
139  /* --- DEFAULT 1-rarefaction and 2-rarefaction --- */
140  soln = 3;
141

143  * the left and right densities (should only happen when left and
144  * right densities differ only slightly and intermediate density
145  * calculated in any step has significant truncation and/or roundoff
146  * errors), default to rarefactions on both left and right */
147  dm = zl*zr*exp((Wl.Vx-Wr.Vx)/(2.0*Iso_csound));
148

161  */

190

231  /* To right of shock */

233  pF->Mx = Ur.Mx*(Wr.Vx) + Wr.d*Iso_csound2;
234  pF->My = Ur.My*(Wr.Vx);
235  pF->Mz = Ur.Mz*(Wr.Vx);

295  return 0.0;

298  if (fh == 0.0) return x2;

312  || (fabs(2.0*f) > fabs(dxold*df))) {
313  dxold=dx;
314  dx=0.5*(xh-xl);
315  rts=xl+dx;
316  if (xl == rts) return rts;
317  } else {
318  dxold=dx;
319  dx=f/df;
320  temp=rts;
321  rts -= dx;
322  if (temp == rts) return rts;
323  }

325  (*funcd)(rts,vl,vr,dmin,dmax,&f,&df);
326  if (f < 0.0)
327  xl=rts;
328  else
329  xh=rts;
330  }
331  /* Cap the number of iterations but don't fail */
332  return rts;

347 static Real guessP(const Prim1DS Wl, const Prim1DS Wr)
348 {
349 
350  Real al = sqrt((double) Gamma*Wl.P/Wl.d); /* left sound speed */
351  Real ar = sqrt((double) Gamma*Wr.P/Wr.d); /* right sound speed */
352  Real ppv; /* inital guess for pressure */
353  Real gl, gr, p_0;
354  Real TOL = 1.0e-6;
355 
356  ppv = 0.5*(Wl.P + Wr.P)-0.125*(Wr.Vx-Wl.Vx)*(Wl.d+Wr.d)*(al+ar);
357 
358  if (ppv < 0.0)
359  ppv = 0.0;
360 
361  gl = sqrt((double) (2.0/(Wl.d*(Gamma+1)))/
362  ((Gamma-1)*Wl.P/(Gamma+1) + ppv));
363  gr = sqrt((double) (2.0/(Wr.d*(Gamma+1)))/
364  ((Gamma-1)*Wr.P/(Gamma+1) + ppv));
365  p_0 = (gl*Wl.P + gr*Wr.P - (Wr.Vx-Wl.Vx))/(gr + gl);
366 
367  if (p_0 < 0.0)
368  p_0 = TOL;
369 
370  return p_0;
371 }

377 static Real PFunc(const Prim1DS W, Real POld)
378 {
379  Real f;
380  Real a = sqrt((double) Gamma*W.P/W.d);
381  Real Ak, Bk;
382 
383  /* rarefaction wave */
384  if (POld <= W.P)
385  f = 2*a/(Gamma-1) * (pow((double) POld/W.P, (double)
386  (Gamma-1)/(2*Gamma)) - 1);
387 
388  /* shock wave */
389  else {
390  Ak = 2/(W.d*(Gamma+1));
391  Bk = W.P*(Gamma-1)/(Gamma+1);
392 
393  f = (POld - W.P) * sqrt((double) Ak/(POld + Bk));
394  }
395 
396  return f;
397 }

433 static Real getPC(const Prim1DS Wl, const Prim1DS Wr)
434 {
435  Real POld;
436  Real VxDiff = Wr.Vx - Wl.Vx;
437  int i = 0;
438  Real TOL = 1.0e-6;
439  Real change, p, fr, fl, frder, flder;
440  Real MAX_ITER = 100;
441 
442  POld = guessP(Wl, Wr);
443 
444  while (i < MAX_ITER) {
445 
446  fl = PFunc(Wl, POld);
447  fr = PFunc(Wr, POld);
448  flder = PFuncDeriv(Wl, POld);
449  frder = PFuncDeriv(Wr, POld);
450 
451  p = POld - (fl + fr + VxDiff)/(flder + frder);
452 
453  change = 2.0*fabs((p-POld)/(p+POld));
454  if (change <= TOL)
455  return p;
456 
457  if (p < 0.0)
458  p = TOL;
459 
460  POld = p;
461  i++;
462  }
463 
464  /* exit failure due to divergence */
465  ath_error("[exact_flux]: Divergence in Newton-Raphson \
466 iteration: p = %e\n", p);
467 
468 }
