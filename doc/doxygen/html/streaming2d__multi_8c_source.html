484 void MultiNSH(int n, Real *tstop, Real *mratio, Real etavk,
485  Real *uxNSH, Real *uyNSH, Real *wxNSH, Real *wyNSH)
486 {
487  int i,j;
488  Real *Lambda1,**Lam1GamP1, **A, **B, **Tmp;
489 
490  Lambda1 = (Real*)calloc_1d_array(n, sizeof(Real)); /* Lambda^{-1} */
491  Lam1GamP1=(Real**)calloc_2d_array(n, n, sizeof(Real));/* Lambda1*(1+Gamma) */
492  A = (Real**)calloc_2d_array(n, n, sizeof(Real));
493  B = (Real**)calloc_2d_array(n, n, sizeof(Real));
494  Tmp = (Real**)calloc_2d_array(n, n, sizeof(Real));
495 
496  /* definitions */
497  for (i=0; i<n; i++){
498  for (j=0; j<n; j++)
499  Lam1GamP1[i][j] = mratio[j];
500  Lam1GamP1[i][i] += 1.0;
501  Lambda1[i] = 1.0/(tstop[i]+1.0e-16);
502  for (j=0; j<n; j++)
503  Lam1GamP1[i][j] *= Lambda1[i];
504  }
505 
506  /* Calculate A and B */
507  MatrixMult(Lam1GamP1, Lam1GamP1, n,n,n, Tmp);
508  for (i=0; i<n; i++) Tmp[i][i] += 1.0;
509  InverseMatrix(Tmp, n, B);
510  for (i=0; i<n; i++)
511  for (j=0; j<n; j++)
512  B[i][j] *= Lambda1[j];
513  MatrixMult(Lam1GamP1, B, n,n,n, A);
514 
515  /* Obtain NSH velocities */
516  *uxNSH = 0.0; *uyNSH = 0.0;
517  for (i=0; i<n; i++){
518  wxNSH[i] = 0.0;
519  wyNSH[i] = 0.0;
520  for (j=0; j<n; j++){
521  wxNSH[i] -= B[i][j];
522  wyNSH[i] -= A[i][j];
523  }
524  wxNSH[i] *= 2.0*etavk;
525  wyNSH[i] *= etavk;
526  *uxNSH -= mratio[i]*wxNSH[i];
527  *uyNSH -= mratio[i]*wyNSH[i];
528  wyNSH[i] += etavk;
529  }
530 
531  free(Lambda1);
532  free_2d_array(A); free_2d_array(B);
533  free_2d_array(Lam1GamP1); free_2d_array(Tmp);
534 
535  return;
536 }
