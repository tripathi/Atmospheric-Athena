677  ath_pout(0,"\nterminating on cycle limit\n");
678 #ifdef MPI_PARALLEL
679  else if(use_wtlim && iquit == 103)
680  ath_pout(0,"\nterminating on wall-time limit\n");
681 #endif /* MPI_PARALLEL */
682  else
683  ath_pout(0,"\nterminating on time limit\n");
684 
685 /* Get time used */
686 
687  gettimeofday(&tve,NULL);
688  if(have_times > 0) {
689  times(&tbuf);
690  time1 = tbuf.tms_utime + tbuf.tms_stime;
691  cpu_time = (time1 > time0 ? (double)(time1 - time0) : 1.0)/
692  (double)clk_tck;
693  } else {
694  time1 = clock();
695  cpu_time = (time1 > time0 ? (double)(time1 - time0) : 1.0)/
696  (double)CLOCKS_PER_SEC;
697  }
698 
699 /* Calculate and print the zone-cycles/cpu-second on this processor */
700 
701  zones = 0;
702  for (nl=0; nl<(Mesh.NLevels); nl++){
703  for (nd=0; nd<(Mesh.DomainsPerLevel[nl]); nd++){
704  if (Mesh.Domain[nl][nd].Grid != NULL) {
705  zones += (Mesh.Domain[nl][nd].Grid->Nx[0])*
706  (Mesh.Domain[nl][nd].Grid->Nx[1])*
707  (Mesh.Domain[nl][nd].Grid->Nx[2]);
708  }
709  }}
710  zcs = (double)zones*(double)((Mesh.nstep) - nstep_start)/cpu_time;
711 
712  ath_pout(0," tlim= %e nlim= %i\n",tlim,nlim);
713  ath_pout(0," time= %e cycle= %i\n",Mesh.time,Mesh.nstep);
714  ath_pout(0,"\nzone-cycles/cpu-second = %e\n",zcs);
715 
716 /* Calculate and print the zone-cycles/wall-second on this processor */
717 
718  cpu_time = (double)(tve.tv_sec - tvs.tv_sec) +
719  1.0e-6*(double)(tve.tv_usec - tvs.tv_usec);
720  zcs = (double)zones*(double)((Mesh.nstep) - nstep_start)/cpu_time;
721  ath_pout(0,"\nelapsed wall time = %e sec.\n",cpu_time);
722  ath_pout(0,"\nzone-cycles/wall-second = %e\n",zcs);
723 
724 /* Calculate and print total zone-cycles/wall-second on all processors */
725 #ifdef MPI_PARALLEL
726  zones = 0;
727  for (nl=0; nl<(Mesh.NLevels); nl++){
728  for (nd=0; nd<(Mesh.DomainsPerLevel[nl]); nd++){
729  zones += (Mesh.Domain[nl][nd].Nx[0])*
730  (Mesh.Domain[nl][nd].Nx[1])*
731  (Mesh.Domain[nl][nd].Nx[2]);
732  }}
733  zcs = (double)zones*(double)(Mesh.nstep - nstep_start)/cpu_time;
734  ath_pout(0,"\ntotal zone-cycles/wall-second = %e\n",zcs);
735 #endif /* MPI_PARALLEL */
736 
737 /* complete any final User work */
738 
739  Userwork_after_loop(&Mesh);
740 
741 /* Final output everything (last argument of data_output = 1) */
742 
743  data_output(&Mesh, 1);
744 
745 /* Free all memory */
746 
747  lr_states_destruct();
748  integrate_destruct();
749  data_output_destruct();
750 #ifdef PARTICLES
751  particle_destruct(&level0_Grid);
752  bvals_particle_destruct(&level0_Grid);
753 #endif
754 #if defined(SHEARING_BOX) || (defined(FARGO) && defined(CYLINDRICAL))
755  bvals_shear_destruct();
756 #endif
757 #if defined(RESISTIVITY) || defined(VISCOSITY) || defined(THERMAL_CONDUCTION)
758  integrate_diff_destruct();
759 #endif
760  par_close();
761 
762 #ifdef MPI_PARALLEL
763  MPI_Finalize();
764 #endif /* MPI_PARALLEL */
765 
766  if(time(&stop)>0) /* current calendar time (UTC) is available */
767  ath_pout(0,"\nSimulation terminated on %s",ctime(&stop));
768 
769  ath_log_close(); /* close the simulation log files */
770 
771  return EXIT_SUCCESS;
772 }

780 void change_rundir(const char *name)
781 {
782 #ifdef MPI_PARALLEL
783 
784  int err=0;
785  int rerr, gerr, my_id, status;
786  char mydir[80];
787 
788  status = MPI_Comm_rank(MPI_COMM_WORLD, &my_id);
789  if(status != MPI_SUCCESS)
790  ath_error("[change_rundir]: MPI_Comm_rank error = %d\n",status);
791 
792  if(name != NULL && *name != '\0'){
793 
794  if(my_id == 0)
795  mkdir(name, 0775); /* May return an error, e.g. the directory exists */
796 
797  MPI_Barrier(MPI_COMM_WORLD); /* Wait for rank 0 to mkdir() */
798 
799  baton_start(MAX_FILE_OP, ch_rundir0_tag);
800 
801  if(chdir(name)){
802  ath_perr(-1,"[change_rundir]: Cannot change directory to \"%s\"\n",name);
803  err = 1;
804  }
805 
806  baton_stop(MAX_FILE_OP, ch_rundir0_tag);
807 
808  /* Did anyone fail to make and change to the run directory? */
809  rerr = MPI_Allreduce(&err, &gerr, 1, MPI_INT, MPI_MAX, MPI_COMM_WORLD);
810  if(rerr) ath_perr(-1,"[change_rundir]: MPI_Allreduce error = %d\n",rerr);
811 
812  if(rerr || gerr){
813  MPI_Abort(MPI_COMM_WORLD, 1);
814  exit(EXIT_FAILURE);
815  }
816  }
817 
818 /* Next, change to the local run directory */
819 
820  sprintf(mydir, "id%d", my_id);
821 
822  baton_start(MAX_FILE_OP, ch_rundir1_tag);
823 
824  mkdir(mydir, 0775); /* May return an error, e.g. the directory exists */
825  if(chdir(mydir)){
826  ath_perr(-1,"[change_rundir]: Cannot change directory to \"%s\"\n",mydir);
827  err = 1;
828  }
829 
830  baton_stop(MAX_FILE_OP, ch_rundir1_tag);
831 
832  /* Did anyone fail to make and change to the local run directory? */
833  rerr = MPI_Allreduce(&err, &gerr, 1, MPI_INT, MPI_MAX, MPI_COMM_WORLD);
834  if(rerr) ath_perr(-1,"[change_rundir]: MPI_Allreduce error = %d\n",rerr);
835 
836  if(rerr || gerr){
837  MPI_Abort(MPI_COMM_WORLD, 1);
838  exit(EXIT_FAILURE);
839  }
840 
841 #else /* Serial job */
842
