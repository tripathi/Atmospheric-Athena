284 Real integrateRaref(const Prim1DS Wa, Real a, Real b)
285 {
286 
287  int i;
288  Real integral, sum, diff, f, xx;
289  Real * x;
290  Real * w;
291  int n = 10; /* number of integration points */
292  Real k, h, vt, G, A;
293  Real dd, ccs2, hh;
294 
295 
296  /* allocate memory for absicssas and weights */
297  x = (Real *)malloc((size_t) ((n+1)*sizeof(Real)));
298  w = (Real *)malloc((size_t) ((n+1)*sizeof(Real)));
299 
300  /* get the abscissas and weights for integration */
301  gauleg(-1.0, 1.0, x, w, n);
302 
303  sum = 0.5 * (b + a);
304  diff = 0.5 * (b - a);
305  integral = 0.0;
306 
307  for (i = 1; i <= n; i++) {
308  xx = diff*x[i] + sum;
309 
310  h = 1.0 + Gamma*Wa.P/((Gamma-1.0)*Wa.d);
311  vt = sqrt((double)(Wa.Vy*Wa.Vy + Wa.Vz*Wa.Vz));
312 
313  G = 1.0/sqrt((double)(1.0 - Wa.Vx*Wa.Vx - Wa.Vy*Wa.Vy - Wa.Vz*Wa.Vz));
314  A = h*G*vt;
315 
316  dd = Wa.d*pow((double)(xx/Wa.P), (double) (1.0/Gamma));
317  ccs2 = Gamma*(Gamma - 1.)*xx / (Gamma*xx + (Gamma - 1.)*dd);
318  hh = 1.0 + xx*Gamma/(dd*(Gamma - 1.0));
319 
320  f = sqrt((double) (hh*hh + A*A*(1.0-ccs2)))/(dd*sqrt((double)ccs2)*(hh*hh + A*A));
321 
322  integral = integral + diff*w[i]*f;
323  }
324 
325  free(x);
326  free(w);
327 
328  return integral;
329 }

336 void getShockVars(const Prim1DS Wa, Real Pb, char dir, Real *pJ, Real *pV_s,
337  Real *pD)
338 {
339  Real sign, ha, Ga;
340  Real db, hb; /* specific enthalpy and density behind wave */
341  Real A, B, C,D;
342  Real d, J, v_s;
343  Real TOL = 1e-5;
344 
345  if (dir == 'L') sign = -1.0;
346  if (dir == 'R') sign = 1.0;
347 
348  /* check if pressure is constant across wave */
349  if (fabs(Wa.P - Pb) <= TOL) {
350  if (pJ != NULL)
351  *pJ = 0.0;
352  if (pD != NULL)
353  *pD = Wa.d;
354  if (pV_s != NULL)
355  *pV_s = Wa.Vx;
356 
357  return;
358  }
359 
360  ha = 1.0 + Gamma*Wa.P/((Gamma-1.0)*Wa.d);
361  Ga = 1.0/sqrt((double)(1.0 - Wa.Vx*Wa.Vx - Wa.Vy*Wa.Vy - Wa.Vz*Wa.Vz));
362 
363  /* calculate specific enthalpy */
364  A = 1.0 + (Gamma - 1.0)*(Wa.P-Pb)/(Gamma*Pb);
365  B = 1.0 - A;
366  C = ha*(Wa.P - Pb)/Wa.d - ha*ha;
367 
368  /*check for unphysical specific enthalpies*/
369  if (C > (B*B/(4.0*A)))
370  ath_error("[exact flux]: Unphysical specific enthalpy in intermediate state");
371 
372  hb = (-B + sqrt((double)(B*B - 4.0*A*C)))/(2.0*A);
373 
374  d = Gamma*Pb/((Gamma - 1.0)*(hb - 1.0));
375  if (pD != NULL)
376  *pD = d;
377 
378  J = sign*sqrt((double)((Pb - Wa.P)/(ha/Wa.d - hb/d)));
379  if (pJ != NULL)
380  *pJ = J;
381 
382  A = Wa.d*Wa.d*Ga*Ga;
383  if (pV_s != NULL)
384  *pV_s = (A*Wa.Vx +sign*fabs((float) J)*
385  sqrt((double)(J*J + A*(1.0 - Wa.Vx*Wa.Vx))))/(A + J*J);
386 
387  return;
388 }

492 Real getP(const Prim1DS Wl, const Prim1DS Wr)
493 {
494  Real vRel, vRS, vSS, vRR;
495  Real pmin, pmax; /* brackets for pressure */
496  Real xacc;
497  Real pc; /* pressure in intermediate state*/
498  float tol, eps;
499 
500  /* 1. Compute the relative x-velocity between Wl and Wr */
501  vRel = (Wl.Vx - Wr.Vx) / (1.0 - Wl.Vx*Wr.Vx);
502 
503  /* 2. Determine the wave pattern by comparing vRel with the limiting values of
504  the relative velocity and bracket pressure in intermed state */
505  vSS = getVlim_2S(Wl, Wr);
506  vRS = getVlim_RS(Wl, Wr);
507 
508  if (vRel <= vRS) {
509  wave = Two_R;
510  pmin = 0.0;
511  if (Wl.P < Wr.P) pmax = Wl.P;
512  else pmax = Wr.P;
513  }
514  else if (vRel > vRS && vRel <= vSS) {
515  if (Wl.P > Wr.P) {
516  wave = RS;
517  pmin = Wr.P;
518  pmax = Wl.P;
519  }
520  else {
521  wave = SR;
522  pmin = Wl.P;
523  pmax = Wr.P;
524  }
525  }
526  else {
527  wave = Two_S;
528  if (Wl.P > Wr.P) pmin = Wl.P;
529  else pmin = Wr.P;
530  pmax = 1000 * 0.5*(Wl.P + Wr.P);
531  }
532 
533  /* 3. compute the machine precision */
534  eps = 1.0f;
535  do eps /= 2.0f;
536  while ((float)(1.0 + (eps/2.0)) != 1.0);
537  xacc = eps*0.5*(pmin + pmax);
538 
539  /* 4. use interval bisection to find p in the intermediate state */
540  pc = rtbis_vel(getDelVRel, Wl, Wr, pmin, pmax, xacc);
541  return pc;
542

826 void gauleg(double x1, double x2, Real x[], Real w[], int n)
827 {
828  int m,j,i;
829  Real z1,z,xm,xl,pp,p3,p2,p1;
830  m=(n+1)/2; /* The roots are symmetric, so we only find half of them. */
831  xm=0.5*(x2+x1);
832  xl=0.5*(x2-x1);
833 
834  for (i=1; i<=m; i++) { /* Loop over the desired roots. */
835  z=cos(3.141592654*(i-0.25)/(n+0.5));
836  /* Starting with the above approximation to the ith root, we enter */
837  /* the main loop of refinement by Newton's method. */
838  do {
839  p1=1.0;
840  p2=0.0;
841  for (j=1;j<=n;j++) { /* Recurrence to get Legendre polynomial. */
842  p3=p2;
843  p2=p1;
844  p1=((2.0*j-1.0)*z*p2-(j-1.0)*p3)/j;
845  }
846  /* p1 is now the desired Legendre polynomial. We next compute */
847  /* pp, its derivative, by a standard relation involving also */
848  /* p2, the polynomial of one lower order. */
849  pp=n*(z*p1-p2)/(z*z-1.0);
850  z1=z;
851  z=z1-p1/pp; /* Newton's method. */
852  } while (fabs(z-z1) > EPS);
853  x[i]=xm-xl*z; /* Scale the root to the desired interval, */
854  x[n+1-i]=xm+xl*z; /* and put in its symmetric counterpart. */
855  w[i]=2.0*xl/((1.0-z*z)*pp*pp); /* Compute the weight */
856  w[n+1-i]=w[i]; /* and its symmetric counterpart. */
857  }
858 }
