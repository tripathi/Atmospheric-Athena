376 double ran2(long int *idum)
377 {
378  int j;
379  long int k;
380  static long int idum2=123456789;
381  static long int iy=0;
382  static long int iv[NTAB];
383  double temp;
384 
385  if (*idum <= 0) { /* Initialize */
386  if (-(*idum) < 1) *idum=1; /* Be sure to prevent idum = 0 */
387  else *idum = -(*idum);
388  idum2=(*idum);
389  for (j=NTAB+7;j>=0;j--) { /* Load the shuffle table (after 8 warm-ups) */
390  k=(*idum)/IQ1;
391  *idum=IA1*(*idum-k*IQ1)-k*IR1;
392  if (*idum < 0) *idum += IM1;
393  if (j < NTAB) iv[j] = *idum;
394  }
395  iy=iv[0];
396  }
397  k=(*idum)/IQ1; /* Start here when not initializing */
398  *idum=IA1*(*idum-k*IQ1)-k*IR1; /* Compute idum=(IA1*idum) % IM1 without */
399  if (*idum < 0) *idum += IM1; /* overflows by Schrage's method */
400  k=idum2/IQ2;
401  idum2=IA2*(idum2-k*IQ2)-k*IR2; /* Compute idum2=(IA2*idum) % IM2 likewise */
402  if (idum2 < 0) idum2 += IM2;
403  j=(int)(iy/NDIV); /* Will be in the range 0...NTAB-1 */
404  iy=iv[j]-idum2; /* Here idum is shuffled, idum and idum2 */
405  iv[j] = *idum; /* are combined to generate output */
406  if (iy < 1) iy += IMM1;
407  if ((temp=AM*iy) > RNMX) return RNMX; /* No endpoint values */
408  else return temp;
409 }
