512 Prim1DS Cons1D_to_Prim1D(const Cons1DS *U, const Real *Bx)
513 {
514  Prim1DS Prim1D;
515  Real Msq, M, ME, Dsq, Gamma_1sq, denom;
516  Real a3, a2, a1, a0;
517  Real i1, i2, i3;
518  Real iR, iS, iT;
519  Real ix1=0.0, iB, iC, v, vOverM;
520 
521  /* Step One: Reduce equations to a quartic polynomial in v */
522 
523  Msq = SQR(U->Mx) + SQR(U->My) + SQR(U->Mz);
524  M = sqrt(Msq);
525 
526  if (fabs(M) < TINY_NUMBER) {
527 
528  v = 0.0; /* if M is zero, then v must be identically zero */
529 
530  } else {
531 
532  ME = M * U->E;
533  Dsq = SQR(U->d);
534 
535  Gamma_1sq = SQR(Gamma_1);
536 
537  denom = 1.0 / (Gamma_1sq * (Msq + Dsq));
538 
539  a3 = (-2.0 * Gamma * Gamma_1 * ME) * denom;
540  a2 = (SQR(Gamma) * SQR(U->E) + 2.0*Gamma_1*Msq - Gamma_1sq*Dsq)*denom;
541  a1 = (-2.0 * Gamma * ME) * denom;
542  a0 = Msq * denom;
543 
544  /* Step Two: Solve the polynomial analytically */
545 
546  i1 = -a2;
547  i2 = a3 * a1 - 4.0 * a0;
548  i3 = 4.0 * a2 * a0 - SQR(a1) - SQR(a3) * a0;
549 
550  iR = (9.0 * i1 * i2 - 27.0 * i3 - 2.0 * SQR(i1) * i1) / 54.0;
551  iS = (3.0 * i2 - SQR(a2)) / 9.0;
552  iT = SQR(iR) + SQR(iS) * iS;
553 
554  /* iT may be negative, but ix1 then adds a complex number and its conjugate,
555  * giving us a real value, so ix1 is real no matter what */
556 
557  if (iT < 0) {
558  ix1 = 2.0*pow(sqrt(iR*iR + iT),(ONE_3RD))*cos(atan2(sqrt(-iT),iR)/3.0)
559  - i1/3.0;
560  } else {
561  ix1 = pow((iR + sqrt(iT)),(ONE_3RD)) + pow((iR - sqrt(iT)),(ONE_3RD))
562  - i1/3.0;
563  }
564 
565  iB = 0.5*(a3 + sqrt(SQR(a3) - 4.0*a2 + 4.0*ix1));
566  iC = 0.5*(ix1 - sqrt(SQR(ix1) - 4.0*a0));
567  v = (-iB + sqrt(SQR(iB) - 4.0*iC))/2.0;
568 
569  }
570 
571  /* Step Three: Resolve primitives with the value of v */
572 
573  /* ensure that v is physical */
574  v = MAX(v, 0.0);
575  v = MIN(v, 1.0 - 1.0e-15);
576 
577  vOverM = v / M;
578 
579  if (fabs(M) < TINY_NUMBER) {
580  vOverM = 0.0;
581  }
582 
583  Prim1D.d = sqrt(1.0 - SQR(v)) * U->d;
584 
585  Prim1D.Vx = U->Mx * vOverM;
586  Prim1D.Vy = U->My * vOverM;
587  Prim1D.Vz = U->Mz * vOverM;
588 
589  Prim1D.P = Gamma_1*
590  ((U->E - U->Mx*Prim1D.Vx - U->My*Prim1D.Vy - U->Mz*Prim1D.Vz) - Prim1D.d);
591 
592  return Prim1D;
593 }

613 Prim1DS Cons1D_to_Prim1D(const Cons1DS *U, const Real *Bx)
614 {
615  Prim1DS Prim1D;
616  Real Bsq = 0.0, Msq = 0.0, S = 0.0, Ssq = 0.0;
617  Real Qp = 0.0, Q = 0.0, Ep = 0.0, E = 0.0, d = 0.0;
618  Real scrh1, scrh2, tmp1, tmp2;
619  Real Usq, Vsq, Gsq, rho, Chi, pgas, ent;
620  Real fQ, dfQ, dQstep;
621 
622  int nr_success;
623  int q_incs;
624 
625  Gamma_1overGamma = Gamma_1/Gamma;
626 
627  /* Calculate Bsq = B^2, Msq = M^2, S = M \dot B, Ssq = S^2 */
628  Bsq = SQR((*Bx)) + SQR(U->By) + SQR(U->Bz);
629  Msq = SQR(U->Mx) + SQR(U->My) + SQR(U->Mz);
630  S = U->Mx * (*Bx) + U->My * U->By + U->Mz * U->Bz;
631  Ssq = SQR(S);
632 
633  /* Assign input energy & density to local variables */
634  E = U->E;
635  d = U->d;
636 
637  /* Starting guess for W, based on taking the +ve */
638  /* root of Eqn. A27, guarantees that p is +ve */
639  scrh1 = -4.0*(E - Bsq);
640  scrh2 = Msq - 2.0*E*Bsq + Bsq*Bsq;
641  Q = ( - scrh1 + sqrt(fabs(scrh1*scrh1 - 12.0*scrh2)))/6.0;
642 
643  nr_success = 0;
644  if (Q < 0.0) {
645  Q = d;
646  } else if (Q != Q) {
647  nr_success = -4;
648  }
649 
650  /* 1d NR algorithm to find W from A1, converges */
651  /* when W'(k+1) / W(k) < tol, or max iterations reached */
652  dQstep = 1.0;
653  q_incs = 0;
654  while (nr_success == 0 && q_incs < 100000) {
655 
656  if (fabs(dQstep) <= tol) nr_success = 1;
657

659  Vsq = calc_vsq (Bsq,Msq,Ssq,Q);
660  if (Vsq != Vsq){
661  nr_success = -3;
662  }
663  Gsq = 1.0/(1.0-Vsq);
664  Chi = calc_chi (d,Vsq,Gsq,Q);
665  rho = d / sqrt(fabs(Gsq));

667 
668  /* Evaluate Eqn. A24, A25 for the total energy density */
669  fQ = calc_func (Q, E, Bsq, Ssq, Vsq, pgas);
670  dfQ = calc_dfunc (Q, Bsq, Msq, Ssq, d, Vsq, Gsq, Chi);
671 
672  /* Check that we didn't get a NaN in the process */
673  /*printf ("Function evaluations %12.6e %12.6e\n",fQ,dfQ);*/
674  if (fQ != fQ) {
675  nr_success = -1;
676  }
677  if (dfQ != dfQ) {
678  nr_success = -2;
679  }
680 
681  /* If we start out very close to the solution try and make sure we don't
682  * overshoot. */
683  if (fabs(fQ) < 0.1 && q_incs == 0) {
684  Q *= 10;
685  Vsq = calc_vsq (Bsq,Msq,Ssq,Q);
686  Gsq = 1.0/(1.0-Vsq);
687  Chi = calc_chi (d,Vsq,Gsq,Q);
688  rho = d / sqrt(fabs(Gsq));
689  pgas = Gamma_1*Chi/Gamma;
690

692  dfQ = calc_dfunc (Q, Bsq, Msq, Ssq, d, Vsq, Gsq, Chi);
693  }
694 
695  /* Calculate step for NR update, check that it's not a Nan, update/check Q */
696  dQstep = fQ / dfQ;
697  if (dQstep != dQstep){
698  nr_success = -2;
699  }
700  Q -= dQstep;
701  if (Q != Q){
702  nr_success = -1;
703  }
704  q_incs ++;
705 
706  /* Rinse and repeat */
707  }
708 
709  /* If we convered (indicated by nr_success = 1) then check solution */
710  if (nr_success == 1){
711  /*Q = MAX(Q, d*(1.0e0+1.0e-6));*/
712 
713  Vsq = calc_vsq (Bsq,Msq,Ssq,Q);
714  Gsq = 1.0/(1.0-Vsq);
715  Chi = calc_chi (d,Vsq,Gsq,Q);
716  rho = d / sqrt(fabs(Gsq));
717  pgas = Gamma_1*Chi/Gamma;
718 
719  if (pgas < 0.0) nr_success = 2;
720  if (Vsq > 1.0) nr_success = 3;
721  if (Vsq < 0.0) nr_success = 4;
722  }
723 
724  if (nr_success == 2) {
725  /* Case of p < 0, in which case we return floor values */
726  /* Should be amended to either solve the cold MHD eqn's, fall back on the
727  * entropy or tell the integrator to switch to a more diffusive update */
728 
729  tmp1 = 1.0 / Q;
730  tmp2 = 1.0 / (Q + Bsq);
731  Prim1D.d = MAX(rho,1.0e-4);
732  Prim1D.P = MAX(pgas,1.0e-5);
733  Prim1D.Vx = (U->Mx + S*(*Bx)*tmp1)*tmp2;
734  Prim1D.Vy = (U->My + S*U->By*tmp1)*tmp2;
735  Prim1D.Vz = (U->Mz + S*U->Bz*tmp1)*tmp2;
736

738  Prim1D.Bz = U->Bz;

741
